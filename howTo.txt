// Import the fetchResource function
// Note: This assumes that the fetchResource function is exported in the module where it is defined.

import { fetchResource } from './apiWrapper';

// Example usage

async function getUser(userId) {
  try {
    const user = await fetchResource(`users/${userId}`);
    console.log('User data:', user);
  } catch (error) {
    console.error('Failed to fetch user data:', error);
  }
}

async function updateUser(userId, userData) {
  try {
    const options = {
      method: 'PUT',
      headers: {
        'Content-Type': 'application/json',
      },
      body: JSON.stringify(userData),
    };

    const updatedUser = await fetchResource(`users/${userId}`, options);
    console.log('Updated user data:', updatedUser);
  } catch (error) {
    console.error('Failed to update user data:', error);
  }
}

// Usage examples

getUser(1);

const updatedUserData = {
  name: 'John Doe',
  email: 'john.doe@example.com',
};

updateUser(1, updatedUserData);
